Google Sheets + Drive setup

Langkah 1: Buka Google Sheet → Extensions → Apps Script

Langkah 2: Tampal kod ini:

function doGet(e){ return out(JSON.stringify(getSheet('Log').getDataRange().getValues())); } function doPost(e){ const d=JSON.parse(e.postData.contents); if(d.action==='add'){ getSheet('Log').appendRow([d.id,d.datetime,d.title,d.category, d.duration,d.tags,d.notes,d.attachUrl,d.attachName,d.claimJson||'']); } else if(d.action==='delete'){ const sh=getSheet('Log'),r=sh.getDataRange().getValues(); for(let i=r.length-1;i>=1;i--)if(r[i][0]==d.id){sh.deleteRow(i+1);break;} } else if(d.action==='update'){ const sh=getSheet('Log'),r=sh.getDataRange().getValues(); for(let i=r.length-1;i>=1;i--)if(r[i][0]==d.id){ sh.getRange(i+1,1,1,10).setValues([[d.id,d.datetime,d.title, d.category,d.duration,d.tags,d.notes,d.attachUrl,d.attachName,d.claimJson||'']]); break; } } else if(d.action==='addClaim'){ const sh=getSheet('Tuntutan'); sh.appendRow([d.logId,d.tarikh,d.nama,d.tujuan,d.dari,d.ke, d.jarak,d.tollP,d.tollB,d.parking,d.hotel,d.pelbagai,d.jumlah, d.bertolakP,d.tibaP,d.bertolakB,d.tibaB,new Date()]); } else if(d.action==='upload'){ const fold=DriveApp.getFoldersByName('PKPJ Logbook').hasNext() ?DriveApp.getFoldersByName('PKPJ Logbook').next() :DriveApp.createFolder('PKPJ Logbook'); const file=fold.createFile( Utilities.newBlob(Utilities.base64Decode(d.data),d.mimeType,d.filename)); file.setSharing(DriveApp.Access.ANYONE_WITH_LINK,DriveApp.Permission.VIEW); return out(JSON.stringify({url:file.getUrl()})); } return out('ok'); } function getSheet(name){ const ss=SpreadsheetApp.getActiveSpreadsheet(); let sh=ss.getSheetByName(name); if(!sh){ sh=ss.insertSheet(name); if(name==='Log'){ const h=['ID','Datetime','Tajuk Tugas','Kategori','Tempoh (jam)', 'Tag','Nota','URL Lampiran','Nama Lampiran','Tuntutan Claim']; sh.appendRow(h); const hdr=sh.getRange(1,1,1,h.length); hdr.setBackground('#0C447C').setFontColor('#FFFFFF') .setFontWeight('bold').setFontSize(11); sh.setFrozenRows(1); sh.setColumnWidths(1,10,[80,140,220,130,80,120,200,200,140,300]); } if(name==='Tuntutan'){ const h=['Log ID','Tarikh','Nama Pegawai','Tujuan','Dari','Ke', 'Jarak (km)','Toll Pergi','Toll Balik','Parking','Hotel', 'Belanja Pelbagai','Jumlah Tuntutan','Bertolak (Pergi)', 'Tiba (Pergi)','Bertolak (Balik)','Tiba (Balik)','Dicatat Pada']; sh.appendRow(h); const hdr=sh.getRange(1,1,1,h.length); hdr.setBackground('#854F0B').setFontColor('#FFFFFF') .setFontWeight('bold').setFontSize(11); sh.setFrozenRows(1); sh.setColumnWidths(1,18,[80,90,140,200,200,200,80,80,80,80,80,100,110,100,100,100,100,130]); } } return sh; } function out(s){ return ContentService.createTextOutput(s) .setMimeType(ContentService.MimeType.JSON); }
