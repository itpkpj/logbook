/* ── Paste this ENTIRE code into Apps Script, then redeploy ── */ function doGet(e) { return out(JSON.stringify( getSheet('Log').getDataRange().getValues() )); } function doPost(e) { const d = JSON.parse(e.postData.contents); if (d.action === 'add') { getSheet('Log').appendRow([d.id,d.datetime,d.title,d.category, d.duration,d.tags,d.notes,d.attachUrl,d.attachName, d.claimJson||'',d.otMula||'',d.otTamat||'',d.otDur||'']); } else if (d.action === 'delete') { const sh=getSheet('Log'), r=sh.getDataRange().getValues(); for (let i=r.length-1;i>=1;i--) if (r[i][0]==d.id) { sh.deleteRow(i+1); break; } } else if (d.action === 'update') { const sh=getSheet('Log'), r=sh.getDataRange().getValues(); for (let i=r.length-1;i>=1;i--) if (r[i][0]==d.id) { sh.getRange(i+1,1,1,13).setValues([[d.id,d.datetime,d.title, d.category,d.duration,d.tags,d.notes, d.attachUrl,d.attachName,d.claimJson||'', d.otMula||'',d.otTamat||'',d.otDur||'']]); break; } } else if (d.action === 'addClaim') { getSheet('Tuntutan').appendRow([d.logId,d.tarikh,d.tujuan, d.dari,d.ke,d.jarak,d.tollP,d.tollB,d.parking,d.hotel, d.pelbagai,d.jumlah,d.bertolakP,d.tibaP,d.bertolakB,d.tibaB,new Date()]); } else if (d.action === 'upload') { const fold = DriveApp.getFoldersByName('PKPJ Logbook').hasNext() ? DriveApp.getFoldersByName('PKPJ Logbook').next() : DriveApp.createFolder('PKPJ Logbook'); const file = fold.createFile( Utilities.newBlob(Utilities.base64Decode(d.data),d.mimeType,d.filename)); file.setSharing(DriveApp.Access.ANYONE_WITH_LINK,DriveApp.Permission.VIEW); return out(JSON.stringify({url:file.getUrl()})); } return out('ok'); } function getSheet(name) { const ss = SpreadsheetApp.getActiveSpreadsheet(); let sh = ss.getSheetByName(name); if (!sh) { sh = ss.insertSheet(name); if (name === 'Log') { const h = ['ID','Datetime','Tajuk Tugas','Kategori','Tempoh (jam)', 'Tag','Nota','URL Lampiran','Nama Lampiran','Tuntutan Claim', 'OT Mula','OT Tamat','OT (jam)']; sh.appendRow(h); sh.getRange(1,1,1,h.length) .setBackground('#0C447C').setFontColor('#FFFFFF') .setFontWeight('bold').setFontSize(11); sh.setFrozenRows(1); sh.setColumnWidths(1,13,[80,140,220,130,80,120,200,200,140,300,80,80,70]); } if (name === 'Tuntutan') { const h = ['Log ID','Tarikh','Tujuan','Dari','Ke', 'Jarak (km)','Toll Pergi','Toll Balik','Parking','Hotel', 'Belanja Pelbagai','Jumlah Tuntutan','Bertolak (Pergi)', 'Tiba (Pergi)','Bertolak (Balik)','Tiba (Balik)','Dicatat Pada']; sh.appendRow(h); sh.getRange(1,1,1,h.length) .setBackground('#854F0B').setFontColor('#FFFFFF') .setFontWeight('bold').setFontSize(11); sh.setFrozenRows(1); sh.setColumnWidths(1,17,[80,90,200,200,200,80,80,80,80,80,100,110,100,100,100,100,130]); } } return sh; } function out(s) { return ContentService.createTextOutput(s) .setMimeType(ContentService.MimeType.JSON); }
⚠️ Jika Sheet Log sudah wujud — tambah 3 kolum baharu secara manual:
Kolum K = OT Mula  |  Kolum L = OT Tamat  |  Kolum M = OT (jam)

⚠️ Deployment settings yang WAJIB betul:

Execute as → Me (your Gmail)
Who has access → Anyone (bukan "Anyone with Google account")
Selepas deploy → klik Authorize access → Allow semua

Langkah 3: Deploy → New deployment → settings seperti atas → Copy URL → tampal di Connect bar → tekan Connect.
